cmake_minimum_required(VERSION 3.0)

project(
    yaya_logtime
    VERSION      0.1.0.0
    LANGUAGES    C
    HOMEPAGE_URL "https://github.com/seityaya/c_logtime"
    DESCRIPTION  "Tree view time logger"
)

set(SRC_LIST
    yaya_logtime.c
)

    yaya_logtime.h
)

target_link_libraries(${PROJECT_NAME} PUBLIC yaya_memory)
